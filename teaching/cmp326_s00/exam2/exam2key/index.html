import java.awt.*; import java.awt.event.*;
public class Lab4 extends Frame implements ActionListener
{   public static final int RADIUS = 75;
    public static final int XCENTER = 150;
    public static final int YCENTER = 125;
    public static void main(String[] args)
    {   Lab4 pieDemo  = new Lab4();
        pieDemo.setVisible(true);
    }
    public Lab4()
    {   setLayout(new BorderLayout());
        setSize(300, 200);
        addWindowListener(new WindowDestroyer());
        textPanel = new Panel(); 
        Label textLabel = new Label("Enter number of pieces:");
        textPanel.add(textLabel);
        theText = new TextField(3);
        textPanel.add(theText);
        Button doneButton = new Button("Done");
        doneButton.addActionListener(this);
        textPanel.add(doneButton);
        add(textPanel, "South");
    }
    public void actionPerformed(ActionEvent e) 
    {   if (e.getActionCommand().equals("Done"))
//CHANGES: ADDED TRY AND CATCH BLOCKS:
        { try{
            pieces = 
                Integer.valueOf(theText.getText().trim()).intValue();
          }
          catch (Exception e)
          {
            theText.setText("Error: Reenter number.");
          } 
        }
//END OF CHANGES
        else
            theText.setText("Error in memo interface");
        repaint();
    } 
    public void paint(Graphics g)
    {   g.setColor(Color.black);
        g.drawOval(XCENTER-RADIUS, YCENTER-RADIUS, 2*RADIUS, 2*RADIUS); 
        for ( int i = 0 ; i < pieces ; i++ )
