Problem: Write a program that takes the take-off and landing times of a flight and prints out the length of the flight.
Algorithm (first draft):
  1. Get input from user.
  2. Calculate length of flight
  3. Print out results.
Pseudocode:
(half way between English and C++): 
main()
{
    int thour, tmin,	// take-off hour and minute
    	lhour, lmin,	// landing hour and minute
    	fhour, fmin;	// length of flight, hour and minute
    	
    get_input ( thour, tmin, lhour, lmin );
    calculate_length ( thour, tmin, lhour, lmin, fhour, fmin );
    print_length ( fhour, fmin );
}

#include < iostream.h >

// Program that calculates length of flight
//
// Jean Potatohead
// jpotatoh@scudc.scu.edu
//
// This program takes the take-off and landing 
// times of a flight and prints the length of
// the flight.

void get_input ( int& thour, int& tmin, int& lhour, int& lmin );
// This function prompts the user for the flight times.
// It reads in the hours and minutes into separate integer
// variables and assumes hours are entered following the
// 24 hour clock.  This assumption is to avoid dealing with
// AM and PM.

void calculate_length ( int thour, int tmin, int lhour, 
	int lmin, int& fhour, int& fmin );
// This function takes the flight times and calculates the
// length of the flight.  The length is then stored in 
// fhour and fmin

void print_length ( int fhour, int fmin );
// This function prints the flight length stored in 
// fhour and fmin.

main()
{
    int thour, tmin,	// take-off hour and minute
    	lhour, lmin,	// landing hour and minute
    	fhour, fmin;	// length of flight, hour and minute
    	
    get_input ( thour, tmin, lhour, lmin );
    calculate_length ( thour, tmin, lhour, lmin, fhour, fmin );
    print_length ( fhour, fmin );
}

void get_input ( int& thour, int& tmin, int& lhour, int& lmin )
{
    cout << "Please enter the hour the flight departed\n"
    	<< "(using the 24-hour clock): ";
    cin >> thour;
    cout << "Please enter the minute the flight departed: ";
    cin >> tmin;

    cout << "Please enter the hour the flight arrived\n"
    	<< "(using the 24-hour clock): ";
    cin >> lhour;
    cout << "Please enter the minute the flight arrived: ";
    cin >> lmin;

    return;
}

void calculate_length ( int thour, int tmin, int lhour, 
	int lmin, int& fhour, int& fmin )
{
    int t_totalmin, l_totalmin, f_totalmin;
    
    if ( thour > lhour )	// Take-off and landing on different days
    {
    	lhour = lhour + 24;
    }
    // Convert to minutes to make subtraction easier:
    t_totalmin = thour*60 + tmin;
    l_totalmin = lhour*60 + lmin;
    f_totalmin = l_totalmin - t_totalmin;
	
    // Convert back to hours and minutes:
    fhour = f_totalmin / 60;
    fmin = f_totalmin % 60;

    return;
}

void print_length ( int fhour, int fmin )
{
    cout << "The length of the flight is "
         << fhour << " hours and "
         << fmin << " minutes.\n\n";
    return;
}
