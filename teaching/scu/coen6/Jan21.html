float convert( float miles_par )
// Converts miles_par to kilometers by multiplying by 1.6
// Returns the kilometers as a float.
{
    float km_local;

    km_local = miles_par * 1.6;
    return( km_local );
}

Problem: Write a program that gets 2 numbers, stores them in variables num1 and num2 and then swaps in which variable each is stored.
Algorithm:
  1. Get input from user.
  2. Swap values for variables.
  3. Print out results.
Pseudocode:
(half way between English and C++): 
main()
{
    int num1, num2;

    getinput( num1, num2 );
    swap( num1, num2 );
    showresults( num1, num2);
}

#include < iostream.h >

void getinput ( int& input1, int& input2 );
// Reads 2 numbers from the keyboard

void swap ( int& var1, int& var2 );
// Swaps the values of var1 and var2

void showresults ( int out1, int out2 );
// Print out the result of the swap

main()
{
    int num1, num2;

    getinput( num1, num2 );
    swap( num1, num2 );
    showresults( num1, num2);
}

void getinput ( int& input1, int& input2 )
// Reads 2 numbers from the keyboard
{
    cout << "Please enter the first number: ";
    cin >> input1;
    cout << "Please enter the second number: ";
    cin >> input2;
    return;
}

void swap ( int& var1, int& var2 )
// Swaps the values of var1 and var2
{
    int temp;

    temp = var1;
    var1 = var2;
    var2 = temp;

    return;
}

void showresults ( int out1, int out2 )
// Print out the result of the swap
{
    cout << "The first number is " << out1 
         << ".\n";
    cout << "The second number is " << out2 
         << ".\n";
    return;
}
