#include <iostream.h>

// Reads in 10 homework scores into an array and prints out
// the average of the scores.
//
// Jean Potatohead
// jpotatoh@scudc.scu.edu


main()
{
    int hw[10],        // a list of 10 whole numbers
        count = 0;     // used to index the array
        total = 0;     // homework total

    while ( count < 10 ) 
    {
         cout << "Please enter score for homework " 
              << count+1 << ":";
         cin >> hw[count]; 
                       // Why "count+1"?
         cout << endl;
         count++;
    } 

    count = 0;         // reset count to use for totaling

    while ( count < 10 ) 
    {
         total = total + hw[count]; 
         count++;
    } 

    cout << "Your average is " << total/10.0;

}
