// Assumes that messy_file and neat_file are streams that are
// connected to open files.
// Reads decimal numbers from messy_stream and outputs decimals
// in fixed point notation of num_after_decimal point digits
// after the decimal point and width field_width.

int main()
{
    ifstream fin;
    ofstream fout;

    
    fin.open("rawdata.dat");	// Open input file
    if ( fin.fail() )		// Check to see if it works
    {
        cout << "Input file opening failed.\n";
        exit(1);
    } 

    fout.open("neat.dat");	// Open output file
    if ( fout.fail() )		// Check to see if it works
    {
        cout << "Output file opening failed.\n";
        exit(1);
    } 

    make_neat(fin, fout, 5, 12);
  
    // Always close file when done:
    fin.close();	
    fout.close();

}

void make_neat(ifstream& messy_file, ofstream& neat_file,
               int num_after_decimalpoint, int field_width)
{
    neat_file.setf(ios::fixed);
    neat_file.setf(ios::showpoint);
    neat_file.precision(num_after_decimalpoint);

    cout.setf(ios::fixed);
    cout.setf(ios::showpoint);
    cout.precision(num_after_decimalpoint);

    double next;

    while ( messy_file >> next ) 
    {
         cout << setw(field_width) << next << endl; 
         neat_file << setw(field_width) << next << endl; 
    } 
}
