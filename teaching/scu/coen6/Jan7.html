    cout << "Please enter the number of miles: ";

#include <iostream.h>
// Sample Program
//
// Jean Pototahead
// jpotatoh@scudc.scu.edu
//
// Program converts the user's input from miles to kilometers.
// It repeats this until the user wants to quit.


main()
{
    float miles,        // stores the miles the user entered
        km;             // stores the conversion to km
    char answer;        // stores the answer about whether to quit

    do
    {
        cout << endl;   // to make the output look nicer
        cout << "Please enter the number of miles: ";
                        // Prompt user for number of miles
        cin >> miles;   // Store the number in miles
        km = miles*1.6; // Multiply by 1.6 to convert to km
        cout << endl;   // Clear the line
        cout << "The number of kilometers is " << km << ".\n\n";
                        // Print out the number of km

        cout << "Do you want to convert another number? \n";
        cout << "Type y for yes, n for no, and then press return: ";
                        // Prompt user for whether to repeat
        cin >> answer;  // Store answer
        cout << endl;   // to make the output look nicer

    } while (answer == 'y');
}
