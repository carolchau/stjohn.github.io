string copy function: 

void string_copy( char target[], const char source[], int target_size )
// Copies as much of the target into the source as will fit.
// Uses functions from string.h
{
    int source_size;
    int i;

    source_size = strlen( source );

    if ( source_size >= target_size )
    {
  	// source is larger than target, so only get enough to fill target
        source_size = target_size - 1; 
    }
	
    // copy, character by character:
    for ( i = 0 ; i < source_size ; i++ )
        target[i] = source[i];
   
    target[i] = '\0';	// end the string
}
