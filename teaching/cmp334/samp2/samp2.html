6.
What is the output of this program? 
title Mystery Program      (mystery.asm)
; This program is a mystery....
.model small
.stack 100h
.data
.code
main proc
    mov ax, @data
    mov ds, ax
    mov bx, 16             ;(For WriteInt: output will be hexadecimal)
    mov cx, 1
    call myst
    call WriteInt          ; Output is:
    mov cx, 2
    call myst
    call WriteInt          ; Output is:
    mov cx, 3
    call myst
    call WriteInt          ; Output is:
    mov cx, 4
    call myst
    call WriteInt          ; Output is:
    mov ax,4C00h
    int 21h
main endp

myst proc
    push bx
    push cx
    mov ax, 1
    cmp cx, 1
    je end
    dec cx
    push ax
    call myst
    pop bx
    add ax, cx
end:
    pop cx
    pop bx
    ret
myst endp
end main
